# Åbn et terminalvindue og naviger hen til den folder, der indeholder ens script

# Lav et virtuelt miljø (her er .venv navnet på den folder, som miljøet oprettes i)

python.exe -m venv .venv

# Aktiver det virtuelle miljø (Linux)

source venv/bin/activate

# Aktiver det virtuelle miljø (Windows)

.venv\Scripts\activate

# Opgrader pip (Linux)

pip install --upgrade pip

# Opgrader pip (Windows)

python -m pip install --upgrade pip
